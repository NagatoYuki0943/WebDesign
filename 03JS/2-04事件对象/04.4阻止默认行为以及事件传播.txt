1 阻止浏览器的默认行为
    e指的是传递过来的参数
    e.preventDefault()


2 阻止事件默认传播(包括冒泡和捕获),直接停止传播,后面的不执行
    e.stopPropagation();

3
    <script>
        var d3 = document.getElementById("div3");
        var d2 = document.getElementById("div2");
        var d1 = document.getElementById("div1");
        var d0 = document.getElementById("div0");

        //true代表捕获阶段触发,从外到内
        //false代表冒泡阶段触发,从内到外 (默认是false)

        d3.addEventListener('click', function() {
            alert(3);
        }, true)

        //阻止默认传播
        d2.addEventListener('click', function(e) {
            alert(2);
            //阻止事件默认传播(包括冒泡和捕获),直接停止传播,后面的不执行
            e.stopPropagation();
        }, true)

        d1.addEventListener('click', function() {
            alert(1);
        }, true)

        d0.addEventListener('click', function() {
            alert(0);
        }, true)
    </script>
