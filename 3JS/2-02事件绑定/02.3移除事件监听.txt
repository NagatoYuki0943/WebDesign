1 行内绑定和动态绑定不能用代码移除,不过事件监听可以通过代码移除

2 第二个参数要写函数的名字,所以要将函数移出来,即使是在内部将匿名函数改成普通函数也没用,removeEventListener依然找不到

  removeEventListener('事件', 方法名);


    <body>
        <div id="d">
            dddd
        </div>

    </body>
    <script>
        var d = document.getElementById("d");

        function f1() {
            console.log(this.innerHTML); // dddd
        }
        d.addEventListener('click', f1);
        d.addEventListener('click', function() {
            console.log(2); //2
        })

        //第二个参数要写函数的名字,所以要将函数移出来
        d.removeEventListener('click', f2); //不会打印dddd了
    </script>
