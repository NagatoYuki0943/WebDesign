1   this可以再构造函数中将属性和方法定义好
    谁实例化这个构造函数,函数内部的this就指向谁

        function Man() {
            //使用this可以再构造函数中将属性和方法定义好
            this.sex = '男';
            this.height = 40;
        }

        var zhaosi = new Man();
        console.log(zhaosi.height); //40
        zhaosi.height = 39;

        var liuneng = new Man();
        console.log(liuneng.height); //40
